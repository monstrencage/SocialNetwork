val selectM : ('b -> int) -> 'a list * 'b list -> 'a
selectM f (l1,l2) renvoie l'élément x de l1 tel que l'élément y de l2 correspondant (de même index) maximise f.
val selectm : ('b -> int) -> 'a list * 'b list -> 'a
selectm f (l1,l2) renvoie l'élément x de l1 tel que l'élément y de l2 correspondant (de même index) minimise f.
val filter : ('b -> bool) -> 'a list * 'b list -> 'a list
filter f (l1,l2) renvoie le même résultat que : List.filter (fun (x,y) -> f y) (List.combine l1 l2).

val chrono : string -> ('a -> 'b) -> 'a -> 'b
chrono s f x calcule f x, écrit le temps écoulé précédé de s, puis renvoie le résultat du calcul.
val chrono_list : (unit -> 'a) list -> unit
chrono_list l effectue List.map (fun f -> f()) l puis écrit le temps écoulé.
